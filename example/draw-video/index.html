Play the video and look at your favicon

    
      var lib = new Favoritos({
        icon: {
          shape: 'rect',
          lineWidth: 10,
        }
      });

      var video = document.getElementById('video');

      video.addEventListener('play', function () {
        function step() {
          lib.drawImage(video)
          requestAnimationFrame(step)
        }
        requestAnimationFrame(step);
      })
