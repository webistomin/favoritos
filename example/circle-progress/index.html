Scroll page and look at your favicon

    
      var lib = new Favoritos({
        icon: {
          shape: 'circle',
        }
      });

      document.addEventListener('scroll', function() {
        var root = document.documentElement;
        var scrollTopInPx = root.scrollTop;
        var pageHeightInPx = root.scrollHeight - root.clientHeight;
        var scrollPercent = (scrollTopInPx / pageHeightInPx) * 100;
        var scrollPercentRounded = Math.round(scrollPercent);

        lib.drawProgressBar(scrollPercentRounded);
      })
